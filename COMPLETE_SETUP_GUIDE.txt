STEP 3: USE IN YOUR TABLES PAGE
=====================================================

In your existing Tables.tsx or create a new one:

import { TableCard } from "@/components/restaurant/TableCard";
import { ReservationForm } from "@/components/restaurant/ReservationForm";
import { useState } from "react";

export const Tables = () => {
  const venueId = "your-venue-id"; // Get from context/auth
  const [selectedTable, setSelectedTable] = useState(null);
  const [lockExpiry, setLockExpiry] = useState(null);

  // Mock data - replace with actual database query
  const tables = [
    { id: "t1", table_number: 1, capacity: 2, status: "available" },
    { id: "t2", table_number: 2, capacity: 4, status: "available" },
    { id: "t3", table_number: 3, capacity: 6, status: "occupied" },
  ];

  return (
    <div className="p-6">
      <h1 className="text-3xl font-bold mb-6">Restaurant Tables</h1>
      
      {/* Table Grid */}
      <div className="grid grid-cols-4 gap-4 mb-8">
        {tables.map((table) => (
          <TableCard
            key={table.id}
            tableId={table.id}
            tableNumber={table.table_number}
            capacity={table.capacity}
            venueId={venueId}
            status={table.status}
            onSelect={(id) => {
              setSelectedTable(tables.find(t => t.id === id));
              setLockExpiry(new Date(Date.now() + 30000)); // 30s lock
            }}
          />
        ))}
      </div>

      {/* Reservation Form Modal */}
      {selectedTable && lockExpiry && (
        <ReservationForm
          tableId={selectedTable.id}
          tableNumber={selectedTable.table_number}
          capacity={selectedTable.capacity}
          venueId={venueId}
          lockExpiresAt={lockExpiry}
          onSuccess={() => {
            alert("✅ Reservation created!");
            setSelectedTable(null);
            setLockExpiry(null);
          }}
          onCancel={() => {
            setSelectedTable(null);
            setLockExpiry(null);
          }}
        />
      )}
    </div>
  );
};
